6. Table with printf
#!/bin/bash

# Variables
read -p "What is your name ?" MY_NAME
read -p "What is your favorite color?" MY_COLOR
# Arrays
NAMES=("Gintaras" "Andrius" "$MY_NAME")
COLORS=("red" "blue" "$MY_COLOR")
#Main
for i in {0..2}; do
printf "%s %s\n" "${NAMES[$i]}" "${COLORS[$i]}"
done
exit 0

7. Math
#!/bin/bash

#Variables
read -p "Enter first number " FIRST_NUMBER
read -p "Enter second number " SECOND_NUMBER
#Main
echo $((FIRST_NUMBER + SECOND_NUMBER))
echo $((FIRST_NUMBER - SECOND_NUMBER))
echo $((FIRST_NUMBER * SECOND_NUMBER))
echo $((FIRST_NUMBER / SECOND_NUMBER))
exit 0

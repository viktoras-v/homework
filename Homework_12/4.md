
services:

    image: mysql
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: admin
    networks:
      - my_net

  phpmyadmin:
    image: phpmyadmin
    restart: always
    ports:
      - 81:80
    environment:
      MYSQL_ROOT_PASSWORD: admin
    networks:
      - my_net

  wordpress:
    depends_on:
      - db
    image: wordpress:latest
    restart: always
    ports:
      - "80:80"
    environment:
      WORDPRESS_DB_HOST: db:3306
      WORDPRESS_DB_USER: root
      WORDPRESS_DB_PASSWORD: admin
      WORDPRESS_DB_NAME: wp
    networks:
      - my_net

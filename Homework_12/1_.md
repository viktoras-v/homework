**# Dockerfile**

FROM python:3.12-slim

WORKDIR /app

RUN pip install flask

RUN pip install redis

COPY ./app.py .

CMD \["python", "app.py"]



**# Docker-compose**

services:

&nbsp; web:

&nbsp;   build: .

&nbsp;   ports:

&nbsp;     - "5000:5000"

&nbsp;   environment:

&nbsp;     - REDIS\_HOST=redis

&nbsp;     - REDIS\_PORT=6379

&nbsp;   depends\_on:

&nbsp;     - redis

&nbsp; redis:

&nbsp;   image: redis:alpine

&nbsp;   container\_name: db

&nbsp;   restart: always

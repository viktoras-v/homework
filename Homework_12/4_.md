**# Docker-compose with WP**

services:

&nbsp; db:

&nbsp;   image: mysql

&nbsp;   restart: always

&nbsp;   environment:

&nbsp;     MYSQL\_ROOT\_PASSWORD: admin

&nbsp;   networks:

&nbsp;     - my\_net



&nbsp; phpmyadmin:

&nbsp;   image: phpmyadmin

&nbsp;   restart: always

&nbsp;   ports:

&nbsp;     - 81:80

&nbsp;   environment:

&nbsp;     MYSQL\_ROOT\_PASSWORD: admin

&nbsp;   networks:

&nbsp;     - my\_net



&nbsp; wordpress:

&nbsp;   depends\_on:

&nbsp;     - db

&nbsp;   image: wordpress:latest

&nbsp;   restart: always

&nbsp;   ports:

&nbsp;     - "80:80"

&nbsp;   environment:

&nbsp;     WORDPRESS\_DB\_HOST: db:3306

&nbsp;     WORDPRESS\_DB\_USER: root

&nbsp;     WORDPRESS\_DB\_PASSWORD: admin

&nbsp;     WORDPRESS\_DB\_NAME: wp

&nbsp;   networks:

&nbsp;     - my\_net

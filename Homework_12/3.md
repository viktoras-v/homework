# APP Dockerfile
ROM python:3.12-slim
WORKDIR /app
RUN pip install flask
RUN pip install redis
COPY ./app.py .
CMD ["python", "app.py"]

# WORKER Dockerfile
FROM alpine:latest
RUN apk add redis
WORKDIR /app
COPY ./worker.sh .
RUN chmod +x worker.sh
CMD ["./worker.sh"]

# WORKER script
#!/bin/sh
while true; do
    JOB=$(redis-cli -h redis BRPOP jobs 5 | tail -n1) # chatgpt

    if [ -n "$JOB" ]; then
        echo "Processed job: $JOB"
    fi
done


# Docker-compose
services:
  web:
    build: .
    ports:
      - "5000:5000"
    environment:
      - REDIS_HOST=redis
      - REDIS_PORT=6379
    depends_on:
      - redis

  redis:
    image: redis:alpine
    container_name: db
    restart: always

  worker:
    build: ./worker
    depends_on:
      - redis
